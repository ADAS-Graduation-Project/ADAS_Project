28void MSYSCFG_vSetEXTIPort(u8_t A_u8LineID, u8_t A_u8PortID)
29{
30 u8_t L_u8Index = A_u8LineID / 4;
31 u8_t L_u8ShiftAmount = A_u8LineID % 4;
32
33 CLR_BITs(MSYSCFG->EXTICR[L_u8Index], 0b1111, L_u8ShiftAmount, 4);
34 SET_BITs(MSYSCFG->EXTICR[L_u8Index], A_u8PortID, L_u8ShiftAmount, 4);
35}

323void MEXTI_vSetTrigger(u8_t A_u8LineID, u8_t A_u8TriggerStatus)
324{
325
326 if (A_u8LineID < EXTI_MAX_EXTI_NUM)
327 {
328
329 switch (A_u8TriggerStatus)
330 {
331 case EXTI_RisingEdge:
332 SET_BIT(MEXTI->RTSR, A_u8LineID);
333
334 CLR_BIT(MEXTI->FTSR, A_u8LineID);
335 break;
336
337 case EXTI_FallingEdge:
338 SET_BIT(MEXTI->FTSR, A_u8LineID);
339
340 CLR_BIT(MEXTI->RTSR, A_u8LineID);
341 break;
342
343 case EXTI_OnChange:
344 SET_BIT(MEXTI->RTSR, A_u8LineID);
345
346 SET_BIT(MEXTI->FTSR, A_u8LineID);
347 break;
348 }
349 }
350}
