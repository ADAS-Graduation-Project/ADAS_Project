30void ANCC_vModeON( void )
31{
32
33 u32_t L_u32SpeedValue = INITIAL_ZERO ;
34 c8_t L_c8RecievedButton = INITIAL_ZERO ;
35
36 HBluetooth_vSendString( "\nNCC Mode ON\n" ) ;
37
38 do
39 {
40
41 HCarMove_vForward( ) ;
42
43 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
44
45 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
46
47 L_c8RecievedButton = HBluetooth_u8ReceiveByte( ) ;
48
49 switch( L_c8RecievedButton )
50 {
51
52 case EXIT_MODE_CHAR:
53
54 HCarMove_vStop( ) ;
55
56 HBluetooth_vSendString( SPEED0_STR ) ;
57
58 HBluetooth_vSendString( "\nNCC Mode Off\n" ) ;
59
60 break ;
61
62 case INC_SPEED_CHAR:
63
64 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
65
66 if( L_u32SpeedValue >= SPEED_100_PERCENT )
67 {
68 L_u32SpeedValue = SPEED_100_PERCENT ;
69
70 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
71 }
72 else
73 {
74 L_u32SpeedValue += SPEED_10_PERCENT ;
75
76 HCarMove_vSpeedRatio( L_u32SpeedValue ) ;
77
78 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
79 }
80
81 break ;
82
83 case DEC_SPEED_CHAR:
84
85 L_u32SpeedValue = HCarMove_u32GetCarSpeed( ) ;
86
87 if( L_u32SpeedValue <= SPEED_0_PERCENT )
88 {
89 L_u32SpeedValue = SPEED_0_PERCENT ;
90
91 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
92 }
93 else
94 {
95 L_u32SpeedValue -= SPEED_10_PERCENT ;
96
97 HCarMove_vSpeedRatio( L_u32SpeedValue ) ;
98
99 AMobApp_vSendSpeedValue( L_u32SpeedValue ) ;
100 }
101
102 break ;
103
104 default: /* Do Nothing */ break ;
105 }
106
107 }while( L_c8RecievedButton != EXIT_MODE_CHAR ) ;
108
109}
