32void HDCM_vInitMotor(DCM_MotorConfiguration *pMotorConfiguration)
33{
34
35 VAR(MGPIOx_ConfigType) DCMotorPin1Init =
36 {
37 .Port = pMotorConfiguration->u8Port,
38 .Pin = pMotorConfiguration->u8Pin1,
39 .Mode = GPIOx_MODE_OUTPUT,
40 .OutputType = GPIOx_PUSHPULL,
41 .OutputSpeed = GPIOx_LowSpeed,
42 .InputType = GPIOx_NoPull};
43
44 VAR(MGPIOx_ConfigType) DCMotorPin2Init =
45 {
46 .Port = pMotorConfiguration->u8Port,
47 .Pin = pMotorConfiguration->u8Pin2,
48 .Mode = GPIOx_MODE_OUTPUT,
49 .OutputType = GPIOx_PUSHPULL,
50 .OutputSpeed = GPIOx_LowSpeed,
51 .InputType = GPIOx_NoPull};
52
53 MGPIOx_vInit(&DCMotorPin1Init);
54 MGPIOx_vInit(&DCMotorPin2Init);
55
56 // Give an initial direction to the DC Motor:
57 switch (pMotorConfiguration->u8Direction)
58 {
59
60 case FORWARD:
61 HDCM_vMoveForward(pMotorConfiguration);
62 break;
63
64 case BACKWARD:
65 HDCM_vMoveBackward(pMotorConfiguration);
66 break;
67
68 /* Default case: stop the motor */
69 default:
70 HDCM_vStopMotor(pMotorConfiguration);
71 break;
72 }
73
74 // Give an initial speed to the DC Motor:
75 HDCM_vMotorSpeedCntrl(pMotorConfiguration, pMotorConfiguration->u32SpeedRatio);
76}

166 break;

169 L_u32SpeedValue = MTIM1_u16GetCaptureReg4Value();
170 break;
