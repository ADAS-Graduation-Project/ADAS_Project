87void HTFT_vInit( void )
88{
89
90 // Set pin directions:
91 MGPIOx_vInit( &TFT_A0 ) ;
92 MGPIOx_vInit( &TFT_RST ) ;
93
94 MSPI_vInit( SPI_SRC, SPI_RELATION, SPI_DATA_DIRECTION ) ;
95
96 MSysTick_vInit( ) ;
97
98 // Rest sequence:
99 HTFT_vResetSequence( ) ;
100
101 // Send SLPOUT CMD (0x11):
102 HTFT_vWriteCMD( SLPOUT_CMD ) ;
103
104 // Wait 15ms:
105 MSysTick_vDelayMilliSec( 15 ) ;
106
107 // Send color mode CMD -> RGB565:
108 HTFT_vWriteCMD( COLOR_MODE_CMD ) ;
109 HTFT_vWriteData( RGB565_CMD ) ;
110
111 // DISPON CMD (0X29):
112 HTFT_vWriteCMD( DISPON_CMD ) ;
113
114}

119void HTFT_vShowImage( u16_t const *A_u16Image, u32_t A_u32ImageSize )
120{
121
122 // Set X position:
123 HTFT_vWriteCMD( XPOSITION_CMD ) ;
124
125 // Xstart: 0, Xend: 127
126 HTFT_vWriteData( 0 ) ;
127 HTFT_vWriteData( 0 ) ;
128 HTFT_vWriteData( 0 ) ;
129 HTFT_vWriteData( 127 ) ;
130
131 // Set Y position:
132 HTFT_vWriteCMD( YPOSITION_CMD ) ;
133
134 // Ystart: 0, Yend: 159
135 HTFT_vWriteData( 0 ) ;
136 HTFT_vWriteData( 0 ) ;
137 HTFT_vWriteData( 0 ) ;
138 HTFT_vWriteData( 159 ) ;
139
140 // Send image data:
141 HTFT_vWriteCMD( MEMORY_WRITE_CMD ) ;
142
143 for( u16_t PX = INITIAL_ZERO; PX < A_u32ImageSize; PX++ )
144 {
145
146 HTFT_vWriteData( A_u16Image[PX] >> 8 ) ; // MSB
147 HTFT_vWriteData( A_u16Image[PX] & 0x00FF ) ; // LSB
148
149 }
150
151}
